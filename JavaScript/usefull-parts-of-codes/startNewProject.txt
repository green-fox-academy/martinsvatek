INICIALIZATION
-----
npm init //package.json
npm install react react-dom --save //dependencies
npm install @babel/core babel-loader @babel/preset-env @babel/preset-react --save-dev //development dependencies
npm install webpack webpack-cli webpack-dev-server --save-dev //development dependencies
npm install eslint --save-dev //development dependencies


CONFIGURE WEBPACK.CONFIG.JS
-----
module.exports = {
  module: {
    rules: [
      {
        test: /\.js$/,
        exclude: /node_modules/,
        loader: 'babel-loader',
      },
    ],
  },
  devServer: {
    contentBase: './dist',
  },
};


CONFIGURE .ESLINTRC.JS
-----
module.exports = {
  "extends": "airbnb",
  "rules": {
    "react/jsx-filename-extension": [
      1,
      {
        "extensions": [
          ".js",
        ],
      },
    ],
  },
  "env": {
    "browser": true,
  },
};


CONFIGURE .BABELRC
-----
{
  "presets": [
    "@babel/env",
    "@babel/react"
  ]
}


SET UP THE FOLLOWING NPM SCRIPTS
-----
"dev": "webpack-dev-server --mode development"
"lint": "eslint ./src/"


CREATE DIST/INDEX.HTML
-----
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>Title</title>
</head>
<body>
  <div id="root"></div>
  <script src="main.js"></script>
</body>
</html>


CREATE SRC/INDEX.JS
-----
import React from 'react';
import ReactDOM from 'react-dom';

/* JSX */
ReactDOM.render( /* what do i want to render, where do i want to render it */
  <h1>Hello, Green Fox Academy!</h1>,
  document.getElementById('root'),
)  


GITIGNORE
-----
node_modules/
dist/*
!dist/index.html


START
-----
npm run dev


END
